16  import gov.usgs.volcanoes.util.args.Args;

19  public class DateRangeArgTest {
20  
21  	private static final String FORMAT = "yyyyMMddHHmm";
22  	private static final String START_TIME = "201508010000";
23  	private static final String END_TIME = "201508020000";
24  
25  	Arguments arg;
26  
27  	@Before
28  	public void setUp() throws JSAPException {
29  		arg = new DateRangeArg(FORMAT, new Args(null, null, new Parameter[0]));
30  	}
31  
32  	@Test
33  	public void when_givenNothing_then_returnNothing() throws JSAPException {
34  		new DateRangeArg(FORMAT, new Args(null, null, new Parameter[0]));
35  	}
36  	
37  	@Test(expected = ParseException.class)
38  	public void when_OnlyStartTime_then_Exception() throws Exception {
39  		String[] commandLine1 = { "--startTime", START_TIME };
40  		arg.parse(commandLine1);
41  	}
42  
43  	@Test(expected = ParseException.class)
44  	public void when_OnlyEndTime_then_Exception() throws Exception {		
45  		String[] commandLine2 = { "--endTime", END_TIME };
46  		arg.parse(commandLine2);
47  	}
48  
49  	@Test(expected = ParseException.class)
50  	public void when_EndTimeNotAfterStartTime_then_Exception() throws Exception {
51  		String[] commandLine2 = { "--startTime", END_TIME, "--endTime", START_TIME };
52  		arg.parse(commandLine2);
53  	}
54  
55  	@Test
56  	public void when_nothingIn_then_nothingOut() throws Exception {
57  		arg.parse(new String[0]);
58  	}
59  	
60  	
61  	@Test
62  	public void when_InputGood_then_OutputGood() throws Exception {
63  		SimpleDateFormat format = new SimpleDateFormat(FORMAT);
64  		format.setTimeZone(TimeZone.getTimeZone("UTC"));
65  		String[] commandLine2 = { "--startTime", START_TIME, "--endTime", END_TIME };
66  		JSAPResult jsapResult = arg.parse(commandLine2);
67  		assertEquals(jsapResult.getDate("startTime"), format.parse(START_TIME));
68  		assertEquals(jsapResult.getDate("endTime"), format.parse(END_TIME));
69  	}
70  }
