6   package gov.usgs.volcanoes.core.args;
7   
8   import com.martiansoftware.jsap.JSAPException;
9   import com.martiansoftware.jsap.JSAPResult;
10  import com.martiansoftware.jsap.Parameter;
11  import com.martiansoftware.jsap.SimpleJSAP;
12  
13  /**
14   * An argument processor.
15   *
16   * @author Tom Parker
17   */
18  public class Args implements Arguments {
19    protected SimpleJSAP jsap;
20    protected JSAPResult jsapResult;
21  
22    /**
23     * Construct a Args instance.
24     *
25     * @param programName Name of the application use is running
26     * @param explanation Brief explanation of application use
27     * @param parameters Prepopulated list of JSAP Parameters
28     * @throws JSAPException if SimpleJSAP throws it
29     */
30    public Args(String programName, String explanation, Parameter[] parameters) throws JSAPException {
31      jsap = new SimpleJSAP(programName, explanation, parameters);
32    }
33  
34    public Parameter getById(String id) {
35      return jsap.getByID(id);
36    }
37  
38    /**
39     * Parse the command line.
40     * 
41     * @param args The command line args
42     * @return the parsed command line
43     */
44    public JSAPResult parse(String[] args) {
45      jsapResult = jsap.parse(args);
46  
47      return jsapResult;
48    }
49  
50    public void registerParameter(Parameter parameter) throws JSAPException {
51      jsap.registerParameter(parameter);
52    }
53  }
