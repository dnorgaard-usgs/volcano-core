4   import org.apache.log4j.Level;
5   import org.apache.log4j.Logger;
6   import org.apache.log4j.PatternLayout;
7   import org.apache.log4j.RollingFileAppender;
8   
9   import java.io.IOException;
10  
11  public class Log {
12    private static String LOG_PATTERN = "%d{yyyy-MM-dd hh:mm:ss} %5p - %m (%F:%L)%n";
13  
14    /**
15     * Add a rolling file appender to the root logger.
16     * 
17     * @param name name of file
18     * @throws IOException when file cannot be created or modified
19     */
20    public static void addFileAppender(String name) throws IOException {
21      PatternLayout layout = new PatternLayout(LOG_PATTERN);
22      RollingFileAppender fileAppender = new RollingFileAppender(layout, name);
23      fileAppender.setMaxFileSize("1MB");
24      fileAppender.setMaxBackupIndex(2);
25      Logger.getRootLogger().addAppender(fileAppender);
26    }
27    
28    /**
29     * Set root logger level.
30     * @param level minimum logging level
31     */
32    public static void setLevel(Level level) {
33      Logger.getRootLogger().setLevel(level);
34    }
35  }
