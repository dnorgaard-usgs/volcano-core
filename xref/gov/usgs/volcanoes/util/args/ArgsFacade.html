26  public class ArgsFacade {
27  	private static final Logger LOGGER = LoggerFactory.getLogger(ArgsFacade.class);
28  
29  	// Everybody needs PROGRAM_NAME, EXPLANATION, and PARAMETERS
30  	public static final String PROGRAM_NAME = "java -jar gov.usgs.volcanoes.args.ArgsFacade";
31  	public static final String EXPLANATION = "I am demonstrate how to use the args package\n";
32  	private static final Parameter[] PARAMETERS = new Parameter[] {
33  			new Switch("knockKnock", 'k', "knockKnock", "Knock Knock.") };
34  
35  	// ConfigFileArg will want a default
36  	public static final String DEFAULT_CONFIG_FILENAME = "facadeConfig.config";
37  
38  	// So will CreateConfigArg
39  	public static final String EXAMPLE_CONFIG_FILENAME = "facadeConfig.config";
40  
41  	public static void main(String... args) throws Exception {
42  
43  		// keep log4j happy
44  		BasicConfigurator.configure();
45  
46  		// base arguments
47  		Arguments arguments = new Args(PROGRAM_NAME, EXPLANATION, PARAMETERS);
48  
49  		// add any decorators that are needed
50  
51  		// config file decorator
52  		arguments = new ConfigFileArg(DEFAULT_CONFIG_FILENAME, arguments);
53  		arguments = new CreateConfigArg(EXAMPLE_CONFIG_FILENAME, arguments);
54  		arguments = new VerboseArg(arguments);
55  
56  		JSAPResult jsapResult = null;
57  		jsapResult = arguments.parse(args);
58  
59  		boolean verbose = jsapResult.getBoolean("verbose");
60  		LOGGER.debug("Setting: verbose={}", verbose);
61  
62  		final String configFileName = jsapResult.getString("config-filename");
63  		LOGGER.debug("Setting: config-filename={}", configFileName);
64  
65  		if (jsapResult.getBoolean("knockKnock"))
66  			System.out.println("Who's there?");
67  	}
68  }
