7   package gov.usgs.volcanoes.util.args.parser;
8   
9   import com.martiansoftware.jsap.ParseException;
10  import com.martiansoftware.jsap.StringParser;
11  
12  import gov.usgs.volcanoes.util.data.Scnl;
13  
14  import java.util.regex.Pattern;
15  
16  /**
17   * Parse a SCNL from a command line argument.
18   *
19   * @author Tom Parker
20   */
21  public class ScnlParser extends StringParser {
22  
23    @Override
24    public Object parse(String arg) throws ParseException {
25      final String[] comps = arg.split(Pattern.quote(Scnl.DELIMITER));
26  
27      Scnl result = null;
28      if (comps.length == 4) {
29        result = new Scnl(comps[0], comps[1], comps[2], comps[3]);
30      } else if (comps.length == 3) {
31        result = new Scnl(comps[0], comps[1], comps[2]);
32      } else {
33        throw new ParseException("Can't parse SCNL: " + arg + " : " + comps.length);
34      }
35  
36      return result;
37    }
38  }
