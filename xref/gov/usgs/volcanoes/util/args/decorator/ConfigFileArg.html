7   import gov.usgs.volcanoes.util.args.ArgsDecorator;

10  /**
11   * Gather a config filename from the command line and place it in
12   * "config-filename". The filename is taken from the first available unflagged
13   * option. Be careful of the order that Args are added.
14   * 
15   * @author Tom Parker
16   * 
17   *         I waive copyright and related rights in the this work worldwide
18   *         through the CC0 1.0 Universal public domain dedication.
19   *         https://creativecommons.org/publicdomain/zero/1.0/legalcode
20   */
21  public class ConfigFileArg extends ArgsDecorator {
22  
23  	/**
24  	 * Constructs a ConfigFileArg and adds its parameter
25  	 * 
26  	 * @param defaultFileName
27  	 *            The value to assign if the parameter is not given on the
28  	 *            command line
29  	 * @param nextArg
30  	 *            The next Arguments object in the chain.
31  	 * @throws JSAPException
32  	 *             If new Parameter cannot be craeted. Not sure why this would
33  	 *             happen
34  	 */
35  	public ConfigFileArg(final String defaultFileName, final Arguments nextArg) throws JSAPException {
36  		super(nextArg);
37  		
38  		registerParameter(new UnflaggedOption("config-filename", JSAP.STRING_PARSER, defaultFileName,
39  				JSAP.NOT_REQUIRED, JSAP.NOT_GREEDY, "The config file name."));
40  	}
41  }
