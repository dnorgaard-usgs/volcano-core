21  public class VerboseArg extends ArgsDecorator {
22  
23  	public VerboseArg(Arguments nextArg) throws JSAPException {
24  		super(nextArg);
25  		nextArg.registerParameter(new Switch("verbose", 'v', "verbose", "Verbose logging."));
26  	}
27  
28  	public JSAPResult parse(String[] args) throws Exception {
29  		JSAPResult jsap = super.parse(args);
30  
31  		if (jsap.getBoolean("verbose"))
32  			Logger.getRootLogger().setLevel(Level.ALL);
33  		return jsap;
34  	}
35  
36  }
